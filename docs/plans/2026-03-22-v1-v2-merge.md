```typescript
/** Map a spawned object's asset name to an emoji from the theme */
export function assetToEmoji(asset: string, theme: GameTheme): string {
  // Known good assets → pick from goodEmojis
  const goodAssets = ['star', 'apple', 'coin', 'heart', 'diamond', 'gift',
    'bubble_red', 'bubble_blue', 'bubble_gold', 'target_normal', 'target_gold', 'target_small'];
  const badAssets = ['bomb', 'meteor', 'ghost', 'obstacle'];

  if (goodAssets.includes(asset)) {
    return theme.goodEmojis[hashString(asset) % theme.goodEmojis.length];
  }
  if (badAssets.includes(asset)) {
    return theme.badEmojis[hashString(asset) % theme.badEmojis.length];
  }
  // Fallback: pick from all emojis
  const all = [...theme.goodEmojis, ...theme.badEmojis];
  return all[hashString(asset) % all.length];
}

function hashString(s: string): number {
  let h = 0;
  for (let i = 0; i < s.length; i++) h = ((h << 5) - h + s.charCodeAt(i)) | 0;
  return Math.abs(h);
}
```

```typescript
// src/engine/renderer/particle-renderer.ts
import { Graphics } from 'pixi.js';
import type { Container } from 'pixi.js';

interface Particle {
  x: number; y: number;
  vx: number; vy: number;
  life: number; maxLife: number;
  color: number; size: number;
}

export class ParticleRenderer {
  private particles: Particle[] = [];
  private graphics: Graphics;

  constructor(container: Container) {
    this.graphics = new Graphics();
    container.addChild(this.graphics);
  }

  /** V1-style burst: radial explosion with gravity */
  burst(x: number, y: number, color: number, count: number = 8): void {
    for (let i = 0; i < count; i++) {
      const angle = Math.random() * Math.PI * 2;
      const speed = 80 + Math.random() * 120;
      this.particles.push({
        x, y,
        vx: Math.cos(angle) * speed,
        vy: Math.sin(angle) * speed,
        life: 0.6 + Math.random() * 0.4,
        maxLife: 1,
        color,
        size: 3 + Math.random() * 3,
      });
    }
  }

  update(dt: number): void {
    const dtSec = dt / 1000;
    this.graphics.clear();

    for (let i = this.particles.length - 1; i >= 0; i--) {
      const p = this.particles[i];
      p.x += p.vx * dtSec;
      p.y += p.vy * dtSec;
      p.vy += 200 * dtSec; // gravity
      p.life -= dtSec;

      if (p.life <= 0) {
        this.particles.splice(i, 1);
        continue;
      }

      const alpha = p.life / p.maxLife;
      this.graphics.circle(p.x, p.y, p.size).fill({ color: p.color, alpha });
    }
  }

  reset(): void {
    this.particles = [];
    this.graphics.clear();
  }
}
```

```typescript
// src/engine/renderer/float-text-renderer.ts
import { Container, Text, TextStyle } from 'pixi.js';

interface FloatText {
  text: Text;
  vy: number;
  life: number;
  maxLife: number;
}

export class FloatTextRenderer {
  private floats: FloatText[] = [];
  private container: Container;

  constructor(container: Container) {
    this.container = container;
  }

  spawn(x: number, y: number, message: string, color: string = '#FFD700'): void {
    const text = new Text({
      text: message,
      style: new TextStyle({ fill: color, fontSize: 28, fontWeight: 'bold', fontFamily: 'Arial' }),
    });
    text.anchor.set(0.5);
    text.position.set(x, y);
    this.container.addChild(text);
    this.floats.push({ text, vy: -80, life: 1.0, maxLife: 1.0 });
  }

  update(dt: number): void {
    const dtSec = dt / 1000;
    for (let i = this.floats.length - 1; i >= 0; i--) {
      const f = this.floats[i];
      f.text.y += f.vy * dtSec;
      f.life -= dtSec;
      f.text.alpha = Math.max(0, f.life / f.maxLife);
      if (f.life <= 0) {
        this.container.removeChild(f.text);
        f.text.destroy();
        this.floats.splice(i, 1);
      }
    }
  }

  reset(): void {
    for (const f of this.floats) {
      this.container.removeChild(f.text);
      f.text.destroy();
    }
    this.floats = [];
  }
}
```
